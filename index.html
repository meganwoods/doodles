flowers.js

Algorithmic doodle flowers that grow around the edges of things — grown from a pencil doodle in a meeting notebook.

usage

<script src="flowers.js"></script>
<script src="cranes.js"></script>
<script src="kites.js"></script>
<script src="heli.js"></script>
<script>
  const field = Flowers.grow('#card', {
    edges: ['top', 'bottom', 'left', 'right'],
    theme: 'autumn',   // or 'noir', 'meadow' — omit for the classic garden
    trigger: 'hover',  // or 'load' (default), 'visible', 'manual'
    density: 1,        // plant count multiplier
    size: [26, 78],    // stem length range, px
    seed: 7,           // set for a reproducible arrangement
  });
  field.regrow();      // fresh arrangement, replay the growth

  const site = Cranes.build('#hero', {
    count: 3,          // a request — packed so they never overlap
    labels: ['FABCO'], // nameplates painted on some decks
    idle: true,        // hooks keep gently swaying after assembly
  });
  site.rebuild();      // a new skyline

  const sky = Kites.fly('#park', { count: 3 });
  sky.relaunch();      // little people, fresh kites

  const pad = Heli.hover('#cta', { labels: ['G-MEGN'] });
  pad.refly();         // fly in again with a new livery
</script>
